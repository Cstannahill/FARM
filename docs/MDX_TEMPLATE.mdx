## Examples

### Complete Working Example

Here's a complete example that demonstrates the key concepts:

```typescript
// apps/api/src/main.py
from farm import FastAPI, MongoDatabase, AIProvider

app = FastAPI()
db = MongoDatabase(url=os.environ["DATABASE_URL"])
ai = AIProvider(provider="ollama", models=["llama3.2"])

@app.post("/api/chat")
async def chat_endpoint(message: str):
    response = await ai.chat(message)
    await db.save_conversation(message, response)
    return {"response": response}
```

```typescript
// apps/web/src/hooks/useChat.ts
import { useMutation } from "@tanstack/react-query";
import { apiClient } from "../lib/api";

export function useChat() {
  return useMutation({
    mutationFn: async (message: string) => {
      const response = await apiClient.post("/api/chat", { message });
      return response.data;
    },
    onSuccess: (data) => {
      console.log("Chat response:", data.response);
    },
  });
}
```

## Troubleshooting

### Common Issues

<Warning title="Database Connection Failed">
  If you see connection errors, check that: - MongoDB is running and accessible
  - The connection string is correct - Network connectivity is available -
  Authentication credentials are valid
</Warning>

**Issue**: `Error: AI model not found`

**Solution**: Make sure the model is downloaded locally:

```bash
ollama pull llama3.2
```

**Issue**: `TypeError: Cannot read property 'data' of undefined`

**Solution**: Check your API client configuration:

```typescript
// Ensure proper error handling
try {
  const response = await apiClient.post("/api/chat", { message });
  return response.data;
} catch (error) {
  console.error("API Error:", error);
  throw error;
}
```
